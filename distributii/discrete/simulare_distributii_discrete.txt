// simularea var. geom. (1)
function Geom1(p)
	k = 0;
	do {
		u = urand();
		k = k + 1;
	} while (u > p);
	return k;
end function



// simularea var. geom. (2)
function Geom2(p)
	u = urand();
	x = int(log(1-u)/log(1-p)) + 1;
	return x;
end function
