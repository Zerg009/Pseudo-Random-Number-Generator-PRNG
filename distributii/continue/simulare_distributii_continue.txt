// simularea var. uniforme pe [a,b)
function Unif(a,b)
	u = urand();
	return a + (b-a)*u;
end function



// simularea prin metoda inversarii
function MetodaInversarii()
	u = urand();
	x = F^-1(u);
	return x;
end function



// simularea var. exponentiale
function SimulExp(theta)
	u = urand();
	x = -theta * log(1-u);
	return x;
end function
